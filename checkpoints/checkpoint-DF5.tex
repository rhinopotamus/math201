\hrulefill

    \item \(h(t)= 16 \, {\left(t + e^{t} + 2\right)}^{4}\)

    \vfill
    \item \(g(x)= 5 \, \Big(\sin\left(x\right)\Big)^{\frac{7}{3}}\)
    \vfill
    \item \(p(y) = 2\ln\left(\sin(y) + y^2 + 3\right)\)
    \vfill
\end{enumerate}
