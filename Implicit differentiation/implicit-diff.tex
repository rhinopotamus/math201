\section{PS7: Implicit differentiation -- \red{Quick reference key}}

\begin{enumerate}[leftmargin=0pt]
    \item (\href{https://activecalculus.org/single/sec-2-7-implicit.html#act-2-7-1}{Activity 2.7.2}) 
    \begin{enumerate}
        \item (anything reasonable)
        \vfill
        \item $\dfrac{dy}{dx} = \dfrac{1}{5y^4 - 15y^2 + 4}$
        \vfill
        \item $\dfrac{dy}{dx} = -\dfrac{1}{6}$, so $L(x) = -\dfrac{1}{6}(x-0)+1$
        \vfill
        \item $y=\pm 1.418697$ and $y=\pm 3.63143$
        \vfill
    \end{enumerate}

    \item (\href{https://activecalculus.org/single/sec-2-7-implicit.html#act-2-7-3}{Activity 2.7.4})
    \begin{enumerate}
        \item $\frac{dy}{dx} = \dfrac{6y-3x^2}{-3y^2 - 6x}$; slope 1; tangent line $y-3 = 1(x+3)$
        \vfill
        \item $\frac{dy}{dx} = \frac{3x^2 + 1}{\cos(y) + 1}$; slope 1/2; tangent line $y-0 = \dfrac{1}{2}(x-0)$
        \vfill
        \item $\frac{dy}{dx} = \frac{3e^{-xy} - 3xye^{-xy}}{3x^2e^{-xy}+2y}$; slope 0.234950; tangent line $y - 1 = 0.234950(x-0.619061)$.
        \vfill
    \end{enumerate}
\end{enumerate}
